var graphoObj = new Grapho( { place: 'example-2' } );

Example 4 - Bar chart

Source code

// Create grapho object, pass a parameter pointing to the destination element
var graphoObj = new Grapho( { place: 'example-4' } );

// Now we'll introduce fillstyle
// fillStyle takes either an string of hex color, ex. '#FF0000'
// or a 'rgba(255,0,0,0.3)' where the first three parameters are
// reg 0-255, green 0-255, blue 0-255, and finally alpha, 0.0-1.0
graphoObj.addDataset({
    type: 'bar',
    fillStyle: 'rgba(0,255,0,0.3)',
    data: [7, 6, 5, 4, 4, 4, 4, 7, 7, 7, 5, 5, 4, 3, 3, 4, 5, 6, 7, 7]
});

Example 5 - Mixing chart types

Source code

var graphoObj5 = new Grapho( { place: 'example-5' } );

// Lets reuse the area chart from a previous example as the first data source
graphoObj5.addDataset({
    type: 'area',
    data: [7, 6, 5, 4, 4, 4, 4, 7, 7, 7, 5, 5, 4, 3, 3, 4, 5, 6, 7, 7, 2, 8, 5, 5, 4, 2]
});

// Then we add the bar chart from the previous example. 
// As this is transparent, the bars below will still be visible
graphoObj5.addDataset({
    type: 'bar',
    fillStyle: 'rgba(0,255,255,0.3)',
    data: [7, 3, 5, 4, 6, 5, 4, 1, 4, 2, 3, 5, 1, 3, 3, 4, 5, 6, 7, 7]
});

// On top of all, we add a red smoothed line chart with custom dots
graphoObj5.addDataset({
    type: 'line',
    strokeStyle: 'rgba(255,127,127,0.7)',
    lineSmooth: false,
    lineDots: true,
    dotSize: 5,
    data: [4,9,8,4,3,4,4,5,4,3,6,7,5,3,4,5,4,5,6,4,2,3,7,2,4]
});

Example 6 - Gap control, non continouos

Source code

var graphoObj = new Grapho( { place: 'example-6' } );

// First, we create a dataset at x positions 0.5, 1.0, 1.5 and 2.0
graphoObj.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    x: { continouos: false },
    data: [ [0.5,1] , [1.0,3] , [1.5,2.5] , [2.0,1] ]
});

// Then, we create a dataset at x positions 5, 5.5, 6 and 6.5
graphoObj.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    data: [ [5,-1] , [5.5,-3] , [6,-2] , [6.5,-3] ]
});

// The default setting of x axises is to be non-continouos
// So the gap won't be shown.

Example 7 - Gap control, continouos

Source code

var graphoObj = new Grapho( { place: 'example-7' } );

// Now we change the x axis settings, so that we get an continouos axis

});

// Now, we'll see the gap

Example 8 - Scatter chart

Source code

var graphoObj = new Grapho( { place: 'example-8' } );

// Ok, now we crate a red scatter chart in non continouos mode, which is generally wrong
graphoObj.addDataset({
    type: 'scatter',
    strokeStyle: '#FF6666',
    x: { continouos: false },
    data: [ 
        [0.5,0.1] , [1.2,1] , [0.5,1.5] , [2.3,3] ,
        [0.6,3.2] , [1.4,2] , [1.5,3.5] , [2.4,4] ,
        [0.7,2.3] , [1.6,3] , [2.5,2.5] , [2.5,5] ,
        [0.8,3.4] , [1.8,1] , [3.5,4.5] , [2.6,2] ,
        [0.9,2.5] , [2.0,2] , [4.5,2.5] , [2.7,3] ,
        [0.0,3.6] , [2.2,3] , [5.5,3.5] , [2.8,4] ,
        [0.1,2.7] , [2.4,1] , [6.5,1.5] , [2.9,2] ,
        [0.2,3.8] , [2.6,2] , [7.5,3.5] , [2.0,3] ,
        [0.3,2.9] , [2.8,3] , [8.5,2.5] , [2.1,4] ,
        [0.4,3.0] , [3.0,1] , [9.5,1.5] , [2.2,2]
    ]

// The red dots are printed with even spaces regardless of the actual distance in x,
// whilst the yellow dots are printed on an continouos axis with their actual distance

Example 9 - Getting crazy wth scales, axises and stuff.

Source code

var graphoObj9 = new Grapho( { place: 'example-9' } );

// The default type is area, if you want to change that, you'll have to specify type, possible values are 
//    line, scatter, area, bar

// Primär (en på primär x, en på primär y)
graphoObj9.addDataset({
    type: 'area',
    x: { scale: true, gridLines: true, name: "Text on primary x axis"}, 
    y: { scale: true, gridLines: true, name: "Text on primary y axis"},
    data: [7, 6, 22, 4, 4, 4, 4, 7, 7, 7, 5, 5, 4, 3, 3, 4, 5, 6, 7, 7, 2, 8, 5, 5, 4, 2]
});

// Sekundär (en sekund primär x, en på på sekundär y)
graphoObj9.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    x: { axis: 2, scale: true, continouos: true, name: "Text on secondary x axis" },
    y: { axis: 2, scale: true, name: "Text on secondary y axis" },
    data: [ [0.5,1] , [1.0,3] , [9.5,39] , [10.0,13] ]
});

// Primär (en på primär x, två på primär y)
graphoObj9.addDataset({
    type: 'area',
    x: { axis: 1, scale: true},
    y: { axis: 3, scale: true, name: "Text on tertiary x axis"},
    data: [7, 6, 5, 4, 4, 4, 4, 7, 7, 7, 5, 5, 4, 3, 3, 4, 5, 6, 7, 7, 2, 8, 5, 5, 4, 2]
});

// Sekundär (två på sekundär x, två på sekundär y)
graphoObj9.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    x: { axis: 4, scale: true, continouos: true, name: "Text on quadiary(?) x axis" },
    y: { axis: 4, scale: true, continouos: true, name: "Text on quadiary(?) y axis" },
    data: [ [0.5,1] , [1.0,3] , [9.5,15] , [10.0,3.3] ]
});

// Sekundär (tre på sekundär x, tre på sekundär y)
graphoObj9.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    x: { axis: 6, scale: true, continouos: true, name: "Text on hexiary(?) x axis" },
    y: { axis: 6, scale: true, continouos: true, name: "Text on hexiary(?) y axis" },
    data: [ [0.5,1] , [1.0,3] , [9.5,13] , [10.0,3.3] ]
});

// Sekundär (tre på sekundär x, fyra på sekundär y)
graphoObj9.addDataset({
    type: 'line',
    lineDots: true,
    lineSmooth: false,
    x: { axis: 1, scale: true, continouos: true },
    y: { axis: 8, scale: true, continouos: true, name: "Text on octiary(?) y axis" },
    data: [ [0.5,1] , [1.0,3] , [9.5,10] , [10.0,3.3] ]
});
