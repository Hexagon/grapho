Example 6 - Add/Remove dataset

Source code

var graphoObj6 = new Grapho( { place: 'example-6' } ),
    currentDS1, currentDS2, currentDS3,
    iteratorSin=0;

function randomData() {

    // Generate random data
    var randomArray1 = [],randomArray2 = [],randomArray3 = [];
    for (var i = 0, l = 40; i < l; i++) {
        randomArray1.push(Math.round(((Math.sin(iteratorSin/10+i/10) * 15)+15)*100)/100);
        randomArray2.push(Math.round((-(Math.cos(iteratorSin/10+i/15+Math.tan(i/5)) * 15)-15)*100)/100);
        randomArray3.push(Math.round(((Math.cos(iteratorSin/10+i/20+Math.sin(i/2)) * 15)+15)*100)/100);
    }
    iteratorSin+=0.7;
    // Remove last dataset, if there is one
    if (currentDS1!==undefined) {
        graphoObj6.removeDataset(currentDS1);
        graphoObj6.removeDataset(currentDS2);
        graphoObj6.removeDataset(currentDS3);
    }
    // Push a new dataset
    currentDS1 = graphoObj6.addDataset({
        type: 'area',
        strokeStyle: '#F06666',
        fillStyle: 'rgba(192,64,64,0.3)',
        y: {  showScale: true, scaleStyle: '#F0F066', showGridLines: true, name: "Primary axis", showLabels: true },
        x: {  showGridLines: true, showLabels: true},
        data: randomArray1
    });
    // Push a new dataset
    currentDS2 = graphoObj6.addDataset({
        type: 'area',
        strokeStyle: '#66F066',
        fillStyle: 'rgba(64,192,64,0.3)',
        y: {  axis:2, showScale: true, scaleStyle: '#F0F066', showLabels: true },
        data: randomArray2
    });
    // Push a new dataset
    currentDS3 = graphoObj6.addDataset({
        type: 'area',
        strokeStyle: '#6666F0',
        fillStyle: 'rgba(64,64,196,0.3)',
        data: randomArray3
    });
    // Iterate
    setTimeout(randomData,1000/60);
}
// Start the loop
randomData();
